let currentQuoteNumber = null; async function loadOrders() { try { const res = await fetch('/api/orders', { credentials: 'include' }); if (res.status === 401) { window.location.href = '/'; return; } const data = await res.json(); allOrders = data.orders || []; renderOrders(allOrders); document.getElementById('orderCount').textContent = `${allOrders.length} order${allOrders.length !== 1 ? 's' : ''} in production`; } catch(e) { document.getElementById('ordersBody').innerHTML = `
Failed to load orders: ${e.message}
`; } } function fmt(n) { return '$' + parseFloat(n||0).toFixed(2).replace(/\B(?=(\d{3})+(?!\d))/g, ','); } function renderOrders(orders) { const tbody = document.getElementById('ordersBody'); if (!orders.length) { tbody.innerHTML = `
📦
No orders in production yet
`; return; } tbody.innerHTML = orders.map(o => { const customer = o.customer_name || '—'; const company = o.company || ''; const foam = o.order_data?.foamColor || '—'; const hinge = o.order_data?.hingePreference || '—'; const isShipped = !!(o.order_data?.shipped?.tracking); const date = o.created_at ? new Date(o.created_at).toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'}) : '—'; const total = o.total ? fmt(o.total) : '—'; const orderUrl = `https://whisperroomquote.up.railway.app/o/${encodeURIComponent(o.quote_number)}`; return `
${o.quote_number}
${customer}
${company ? `
${company}
` : ''} ${foam} ${hinge} ${total} ${isShipped ? `✓ Shipped` : `In Production`} Order ↗ `; }).join(''); } function filterOrders() { const q = document.getElementById('searchInput').value.toLowerCase(); if (!q) { renderOrders(allOrders); return; } const filtered = allOrders.filter(o => (o.quote_number||'').toLowerCase().includes(q) || (o.customer_name||'').toLowerCase().includes(q) || (o.company||'').toLowerCase().includes(q) || (o.deal_name||'').toLowerCase().includes(q) ); renderOrders(filtered); } function openDrawer(quoteNumber) { const order = allOrders.find(o => o.quote_number === quoteNumber); if (!order) return; currentQuoteNumber = quoteNumber; document.getElementById('drawerTitle').textContent = order.quote_number; document.getElementById('drawerSubtitle').textContent = order.deal_name || order.customer_name || ''; // Customer fields const c = order.json_snapshot?.customer || {}; document.getElementById('editFirstName').value = c.firstName || ''; document.getElementById('editLastName').value = c.lastName || ''; document.getElementById('editCompany').value = c.company || order.company || ''; document.getElementById('editEmail').value = c.email || ''; // Address document.getElementById('editAddress').value = c.address || ''; document.getElementById('editCity').value = c.city || ''; document.getElementById('editState').value = c.state || ''; document.getElementById('editZip').value = c.zip || ''; // Order specs const od = order.order_data || {}; setRadio('editFoam', od.foamColor || ''); setRadio('editHinge', od.hingePreference || ''); document.getElementById('editProductionNotes').value = od.productionNotes || ''; document.getElementById('editDeliveryNotes').value = od.deliveryNotes || ''; // Shipment fields const shipped = od.shipped || {}; document.getElementById('editCarrier').value = shipped.carrier || ''; document.getElementById('editTracking').value = shipped.tracking || ''; document.getElementById('editPallets').value = shipped.pallets || ''; document.getElementById('editBoxes').value = shipped.boxes || ''; document.getElementById('editShipDate').value = shipped.date ? new Date(shipped.date).toISOString().split('T')[0] : new Date().toISOString().split('T')[0]; // Shipped badge const badge = document.getElementById('shippedBadge'); if (shipped.carrier && shipped.tracking) { badge.style.display = 'inline-flex'; } else { badge.style.display = 'none'; } // Change log const log = od.changeLog || []; const logSection = document.getElementById('changeLogSection'); const logList = document.getElementById('changeLogList'); if (log.length) { logSection.style.display = 'block'; logList.innerHTML = log.slice().reverse().map(entry => `
${entry.summary} ${new Date(entry.at).toLocaleString('en-US',{month:'short',day:'numeric',hour:'numeric',minute:'2-digit'})}
` ).join(''); } else { logSection.style.display = 'none'; } // Order page link const orderUrl = `https://whisperroomquote.up.railway.app/o/${encodeURIComponent(quoteNumber)}`; document.getElementById('viewOrderBtn').href = orderUrl; // Open drawer document.getElementById('editDrawer').classList.add('open'); document.getElementById('drawerOverlay').classList.add('open'); document.body.style.overflow = 'hidden'; } function closeDrawer() { document.getElementById('editDrawer').classList.remove('open'); document.getElementById('drawerOverlay').classList.remove('open'); document.body.style.overflow = ''; currentQuoteNumber = null; } function setRadio(name, value) { document.querySelectorAll(`input[name="${name}"]`).forEach(inp => { inp.checked = (inp.value === value); }); highlightSel(name); } function highlightSel(name) { document.querySelectorAll(`input[name="${name}"]`).forEach(inp => { const label = inp.closest('.sel-label'); if (!label) return; label.classList.toggle('selected', inp.checked); }); } async function saveOrder() { if (!currentQuoteNumber) return; const btn = document.getElementById('saveBtn'); btn.disabled = true; btn.textContent = 'Saving…'; const order = allOrders.find(o => o.quote_number === currentQuoteNumber); const prevOd = order?.order_data || {}; const prevC = order?.json_snapshot?.customer || {}; const newFoam = document.querySelector('input[name="editFoam"]:checked')?.value || ''; const newHinge = document.querySelector('input[name="editHinge"]:checked')?.value || ''; const newProdNotes = document.getElementById('editProductionNotes').value.trim(); const newDelNotes = document.getElementById('editDeliveryNotes').value.trim(); // Shipment const carrier = document.getElementById('editCarrier').value; const tracking = document.getElementById('editTracking').value.trim(); const shipDate = document.getElementById('editShipDate').value; const pallets = document.getElementById('editPallets').value; const boxes = document.getElementById('editBoxes').value; const isShipping = carrier && tracking; // Customer const customer = { firstName: document.getElementById('editFirstName').value.trim(), lastName: document.getElementById('editLastName').value.trim(), company: document.getElementById('editCompany').value.trim(), email: document.getElementById('editEmail').value.trim(), address: document.getElementById('editAddress').value.trim(), city: document.getElementById('editCity').value.trim(), state: document.getElementById('editState').value.trim(), zip: document.getElementById('editZip').value.trim(), }; // Build change log diff const changes = []; const chk = (label, prev, next) => { if (prev !== next && (prev || next)) changes.push(`${label}: ${prev||'—'} → ${next||'—'}`); }; chk('Foam Color', prevOd.foamColor || '', newFoam); chk('Door Hinge', prevOd.hingePreference || '', newHinge); chk('Production Notes', prevOd.productionNotes || '', newProdNotes); chk('Delivery Notes', prevOd.deliveryNotes || '', newDelNotes); chk('Name', [prevC.firstName,prevC.lastName].filter(Boolean).join(' '), [customer.firstName,customer.lastName].filter(Boolean).join(' ')); chk('Company', prevC.company || '', customer.company); chk('Email', prevC.email || '', customer.email); chk('Address', [prevC.address,prevC.city,prevC.state,prevC.zip].filter(Boolean).join(', '), [customer.address,customer.city,customer.state,customer.zip].filter(Boolean).join(', ')); if (isShipping && !prevOd.shipped?.tracking) changes.push(`Marked Shipped — ${carrier}, ${tracking}`); const payload = { customer, foamColor: newFoam, hingePreference: newHinge, productionNotes: newProdNotes, deliveryNotes: newDelNotes, shipped: isShipping ? { carrier, tracking, date: shipDate, pallets: parseInt(pallets)||0, boxes: parseInt(boxes)||0 } : (prevOd.shipped || null), changes, }; try { const res = await fetch(`/api/orders/${encodeURIComponent(currentQuoteNumber)}`, { method: 'PATCH', headers: { 'Content-Type': 'application/json' }, credentials: 'include', body: JSON.stringify(payload) }); const data = await res.json(); if (data.success) { showToast(isShipping && !prevOd.shipped?.tracking ? '✓ Order marked shipped!' : '✓ Order updated', 'success'); closeDrawer(); loadOrders(); } else { throw new Error(data.error || 'Save failed'); } } catch(e) { showToast('Error: ' + e.message, 'error'); } finally { btn.disabled = false; btn.textContent = 'Save Changes'; } } function showToast(msg, type='') { const t = document.getElementById('toast'); t.textContent = msg; t.className = 'show ' + type; clearTimeout(t._timer); t._timer = setTimeout(() => t.className = '', 3000); } loadOrders();
